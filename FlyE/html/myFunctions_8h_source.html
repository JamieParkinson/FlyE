77 // Uses the very very bad HDF5 library to write to HDF5. Awful code, ugly, impossible to understand. Very fast.
78 void writeHDF5(vector<Particle> &particles, const string &outFilePath,
79  const int &nTimeSteps,
80  const int &nSucceeded, const int &nIonised, const int &nCollided,
81  const int &nParticles) {
82  H5::H5File* outFile = new H5::H5File(outFilePath.c_str(), H5F_ACC_TRUNC); // Initialise data file
83 
84  H5::FloatType dType(H5::PredType::NATIVE_FLOAT); // Type of data (native float)
85  H5::IntType iType(H5::PredType::NATIVE_INT);
86  dType.setOrder(H5T_ORDER_LE); // Little endian
87  iType.setOrder(H5T_ORDER_LE);
88  H5::DSetCreatPropList dPropList; // Properties list
89  hsize_t dChunkDims[4]; // We want out datasets chunked by particle dimensions across time (like our memory vecs)
90  dChunkDims[0] = 1;
91  dChunkDims[1] = 1;
92  dChunkDims[2] = nTimeSteps + 1;
93  dChunkDims[3] = 1;
94  dPropList.setChunk(4, dChunkDims); // Set up chunking - MASSIVE performance increase
95 
96  vector<int> nParticlesOfType;
97  vector<string> typeNames;
98 
99  nParticlesOfType = {nSucceeded, nCollided, nIonised,
100  nParticles - nSucceeded - nCollided - nIonised};
101  typeNames = {"Succeeded", "Collided", "Ionised",
102  "Remaining"};
103 
104  vector<shared_ptr<H5::DataSet> > dSets; // Storing datasets
105  vector<shared_ptr<H5::DataSet> > ntimeDSets; // For neutralisation times
106  vector<shared_ptr<H5::DataSet> > kDSets; // For k-values
107 
108  vector<H5::DataSpace> dSpaces; // Storing dataspaces
109  vector<H5::DataSpace> ntimeDSpaces; // For neutralisation times
110  vector<H5::DataSpace> kDSpaces; // For k-values
111 
112  vector< vector<int> > ntimeVecs(N_TYPES); // For neutralisation time vectors
113  vector< vector<int> > kVecs(N_TYPES); // For k-value vectors
114 
115  vector<int> pTypeCounts = { 0, 0, 0, 0, 0 }; // Particle type counters
116  int pType; // Will be used to select elements of above vectors
117 
118  H5::Group* typeGroup;
119  for (int type = 0; type < N_TYPES; ++type) { // Initialise all my groups and datasets
120  typeGroup = new H5::Group( outFile->createGroup("/" + typeNames[type]));
121 
122  hsize_t dataDims[4]; // I write these in reverse order because it seems that this is how things turn out in MATLAB
123  dataDims[3] = nParticlesOfType[type]; // Particle
124  dataDims[2] = nTimeSteps + 1; // Time
125  dataDims[1] = 2; // Position and velocity
126  dataDims[0] = Physics::N_DIMENSIONS; // x, y, z
127 
128  hsize_t scalarDims[1]; // For neutralisation times and k-values
129  scalarDims[0] = nParticlesOfType[type];
130 
131  dSpaces.emplace_back(4, dataDims);
132  ntimeDSpaces.emplace_back(1, scalarDims);
133  kDSpaces.emplace_back(1, scalarDims);
134 
135  ntimeVecs[type].reserve(nParticlesOfType[type]);
136  kVecs[type].reserve(nParticlesOfType[type]);
137 
138  dSets.emplace_back(
139  make_shared<H5::DataSet>(
140  outFile->createDataSet("/" + typeNames[type] + "/data", dType, dSpaces[type],
141  dPropList))); // Initialise dataset
142  ntimeDSets.emplace_back(
143  make_shared<H5::DataSet>(
144  outFile->createDataSet("/" + typeNames[type] + "/neutralTimes", iType, ntimeDSpaces[type])));
145 
146  kDSets.emplace_back(
147  make_shared<H5::DataSet>(
148  outFile->createDataSet("/" + typeNames[type] + "/ks", iType, kDSpaces[type])));
149  }
150 
151  for (Particle &particle : particles) { // Look through particles
152  if (particle.succeeded()) { // Determine particle type
153  pType = 0;
154  } else if (particle.isDead()) {
155  pType = particle.isDead();
156  } else {
157  pType = N_TYPES - 1;
158  }
159 
160  if (pType == 1 && !storeCollisions) { // If we're discarding collisions
161  continue;
162  }
163 
164  H5::DataSpace fileSpace = dSpaces[pType];
165 
166  // For the following we'll need to change start[1] for pos/vel and start[0] for dimension
167  hyperslabParams thisParams = particle.makeParameters(pTypeCounts[pType]);
168 
169  hsize_t memDims[1];
170  memDims[0] = thisParams.count[2];
171  H5::DataSpace memSpace = H5::DataSpace(1, memDims);
172 
173  for (int d = 0; d < Physics::N_DIMENSIONS; ++d) {
174  thisParams.start[0] = d;
175 
176  thisParams.start[1] = 0; // Position
177  fileSpace.selectHyperslab(H5S_SELECT_SET, thisParams.count,
178  thisParams.start, thisParams.stride);
179  dSets[pType]->write(particle.recallLoc(d).data(), dType, memSpace,
180  fileSpace);
181 
182  thisParams.start[1] = 1; // Velocity
183  fileSpace.selectHyperslab(H5S_SELECT_SET, thisParams.count,
184  thisParams.start, thisParams.stride);
185  dSets[pType]->write(particle.recallVel(d).data(), dType, memSpace,
186  fileSpace);
187  }
188 
189  ++pTypeCounts[pType]; // Increase out count of this particle type
190 
191  ntimeVecs[pType].emplace_back(particle.neutralisationTime()); // Store ntime
192  kVecs[pType].emplace_back(particle.getK()); // Store k
193 
194  particle.forget(); // Clear the memory of the particle we just dealt with
195 
196  cout << &particle - &particles[0] << "/" << nParticles << " exported\r"
197  << flush;
198  }
199 
200  for (int type = 0; type < N_TYPES; ++type) { // Write neutralisation times & k-values
201  ntimeDSets[type]->write(ntimeVecs[type].data(), iType);
202  kDSets[type]->write(kVecs[type].data(), iType);
203  }
204 
205  delete typeGroup;
206  delete outFile;
207 }
